-- logger main ui
function LoggerUI()
   local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

   -- rayfield UI --
   local Window = Rayfield:CreateWindow({
      Name = "Nations roleplay 2 decals logger",
      Icon = 0, -- Icon in Topbar. Can use Lucide Icons (string) or Roblox Image (number). 0 to use no icon (default).
      LoadingTitle = "Starting up...",
      LoadingSubtitle = "by aoaeil",
      ShowText = "Rayfield", -- for mobile users to unhide rayfield, change if you'd like
      Theme = "Default", -- Check https://docs.sirius.menu/rayfield/configuration/themes

      ToggleUIKeybind = "K", -- The keybind to toggle the UI visibility (string like "K" or Enum.KeyCode)

      DisableRayfieldPrompts = true,
      DisableBuildWarnings = false, -- Prevents Rayfield from warning when the script has a version mismatch with the interface

      ConfigurationSaving = {
         Enabled = true,
         FolderName = NR2, -- Create a custom folder for your hub/game
         FileName = "NationsRoleplay2"
      },

      Discord = {
         Enabled = false, -- Prompt the user to join your Discord server if their executor supports it
         Invite = "noinvitelink", -- The Discord invite code, do not include discord.gg/. E.g. discord.gg/ ABCD would be ABCD
         RememberJoins = true -- Set this to false to make them join the discord every time they load it up
      },

      KeySystem = false, -- Set this to true to use our key system
      KeySettings = {
         Title = "Untitled",
         Subtitle = "Key System",
         Note = "No method of obtaining the key is provided", -- Use this to tell the user how to get a key
         FileName = "Key", -- It is recommended to use something unique as other scripts using Rayfield may overwrite your key file
         SaveKey = true, -- The user's key will be saved, but if you change the key, they will be unable to use your script
         GrabKeyFromSite = false, -- If this is true, set Key below to the RAW site you would like Rayfield to get the key from
         Key = {"Hello"} -- List of keys that will be accepted by the system, can be RAW file links (pastebin, github etc) or simple strings ("hello","key22")
      }
   })

   -- functions (IGNORE IT) --
   function flagcopy()
      loadstring(game:HttpGet('https://raw.githubusercontent.com/kelometr/nationsroleplaygames/refs/heads/main/flagcopy.txt'))()
   end
   function leadercopy()
      loadstring(game:HttpGet('https://raw.githubusercontent.com/kelometr/nationsroleplaygames/refs/heads/main/leadercopy.txt'))()
   end
   function autopaint()
      loadstring(game:HttpGet('https://raw.githubusercontent.com/kelometr/nationsroleplaygames/refs/heads/main/autopaint.txt'))()
   end
   function admin()
      loadstring(game:HttpGet('https://raw.githubusercontent.com/EdgeIY/infiniteyield/master/source'))()
   end
   function crashserv()
      loadstring(game:HttpGet(''))()
   end

   -- script --

   -- how to use
   local TabHow = Window:CreateTab("How to use", 0)
   local Paragraph = TabHow:CreateParagraph({Title = "Instructions", Content = "This script copies the link of selected IDs for easy copy-and-paste. Just paste it from clipboard into the chat to copy an ID inside of link"})

   -- logger
   local TabLog = Window:CreateTab("Logger", 0) -- Title, Image
   local Button = TabLog:CreateButton({
      Name = "Copy flag",
      Callback = function()
      flagcopy()
      end,
   })
      local Button = TabLog:CreateButton({
      Name = "Copy leader",
      Callback = function()
      leadercopy()
      end,
   })

   -- creator
   local TabCre = Window:CreateTab("Credits")
   local Label = TabCre:CreateLabel("Script creator: aoaeil", Color3.fromRGB(255, 255, 255), false) -- Title, Icon, Color, IgnoreTheme
   local Label = TabCre:CreateLabel("OP Panel creator: firepalmmy", Color3.fromRGB(255, 255, 255), false) -- Title, Icon, Color, IgnoreTheme

   -- misc
   local TabMisc = Window:CreateTab("Misc")
   local Label = TabMisc:CreateLabel("Other related scripts", Color3.fromRGB(255, 255, 255), false) -- Title, Icon, Color, IgnoreTheme
   local Button = TabMisc:CreateButton({
      Name = "OP Admin Panel (Free autopaint)",
      Callback = function()
      autopaint()
      end,
   })
   local Button = TabMisc:CreateButton({
      Name = "Infinite Yield",
      Callback = function()
      admin()
      end,
   })

   -- crash
   local TabCrash = Window:CreateTab("Auto paint +")
   local Paragraph = TabCrash:CreateParagraph({Title = "Auto Paint +", Content = "Feature available in premium version. Contact @aoaeil in discord for more information (NOT OUT YET)"})
end

LoggerUI()
